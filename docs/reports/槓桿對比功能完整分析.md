def run_leverage_comparison_in_web():
    progress_bar = st.progress(0)
    status_text = st.empty()
    
    leverages = [1, 2, 3, 5, 10, 20]
    total_tests = len(leverages)
    
    results = []
    for i, leverage in enumerate(leverages):
        status_text.text(f"正在測試 {leverage}x 槓桿...")
        progress_bar.progress((i + 1) / total_tests)
        
        # 執行回測
        result = engine.run_single_strategy(strategy, market_data)
        results.append(result)
