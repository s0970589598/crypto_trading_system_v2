## 現狀分析

### ✅ CLI 已完全支持多策略回測

#### 命令格式
```bash
# 多策略回測
python3 cli.py backtest \
  --strategy multi-timeframe-aggressive \
  --strategy breakout-strategy
```

#### 功能特點
1. ✅ 支持多個策略（`--strategy` 可多次指定）
2. ✅ 自動平均分配資金
3. ✅ 顯示每個策略的結果
4. ✅ 顯示組合統計
5. ✅ 保存結果到 JSON

#### 實現代碼
```python
# cli_commands/backtest.py

if len(strategies) == 1:
    # 單策略回測
    result = engine.run_single_strategy(strategy, market_data)
else:
    # 多策略回測
    capital_allocation = {
        strategy.get_id(): 1.0 / len(strategies)
        for strategy in strategies
    }
    
    results = engine.run_multi_strategy(
        strategies,
        market_data,
        capital_allocation
    )
    
    # 打印組合統計
    total_pnl = sum(r.total_pnl for r in results.values())
    print(f"總損益：{total_pnl:.2f} USDT")
```

```python
# web_dashboard_v2.py

elif sub_function == "多策略組合回測":
    st.subheader("📊 多策略組合回測")
    
    # 策略選擇
    st.info("💡 選擇要組合回測的策略")
    
    available_strategies = [
        "multi-timeframe-aggressive",
        "multi-timeframe-relaxed",
        "breakout-strategy"
    ]
    
    selected_strategies = st.multiselect(
        "選擇策略（至少 2 個）",
        available_strategies,
        default=["multi-timeframe-aggressive", "breakout-strategy"]
    )
    
    if len(selected_strategies) < 2:
        st.warning("⚠️ 請至少選擇 2 個策略")
    else:
        if st.button("🚀 執行多策略組合回測", use_container_width=True, type="primary"):
            with st.spinner("正在執行多策略回測，請稍候..."):
                import subprocess
                
                # 構建命令
                cmd = ['python3', 'cli.py', 'backtest']
                for strategy in selected_strategies:
                    cmd.extend(['--strategy', strategy])
                
                result = subprocess.run(
                    cmd,
                    capture_output=True,
                    text=True,
                    timeout=300
                )
                
                if result.returncode == 0:
                    st.success("✅ 多策略回測完成！")
                    st.balloons()
                    with st.expander("查看執行日誌"):
                        st.code(result.stdout[-2000:])
                    st.rerun()
                else:
                    st.error("❌ 執行失敗")
                    st.code(result.stderr)
```
