### 整合方案

```python
import subprocess

def run_backtest_via_cli(strategy_id, capital=1000):
    """通過 CLI 系統運行回測"""
    cmd = [
        'python3', 'cli.py', 'backtest',
        '--strategy', strategy_id,
        '--capital', str(capital)
    ]
    result = subprocess.run(cmd, capture_output=True, text=True)
    return result
```
